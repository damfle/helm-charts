1. Get the application URL by running these commands:

{{- if eq .Values.service.type "ClusterIP" }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace | default "default" }} -l "app.kubernetes.io/name={{ include "llamacpp.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath='{.items[0].metadata.name}')
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace | default "default" }} $POD_NAME -o jsonpath='{.spec.containers[0].ports[0].containerPort}')
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace | default "default" }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace | default "default" }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "llamacpp.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace | default "default" }} -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "Visit http://$NODE_IP:$NODE_PORT to use your application"
{{- end }}
