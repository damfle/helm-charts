1. Get the application URL by running these commands:
{{- if .Values.panel.ingress.enabled }}
{{- range $host := .Values.panel.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.panel.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.panel.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "pterodactyl.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.panel.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch its status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "pterodactyl.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "pterodactyl.fullname" . }} --template "{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}")
  echo http://$SERVICE_IP:{{ .Values.panel.service.port }}
{{- else if contains "ClusterIP" .Values.panel.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "pterodactyl.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

{{- if .Values.mariadb.enabled }}
2. MariaDB is deployed as a separate pod in the same namespace.
   - Service: {{ include "pterodactyl.fullname" . }}-mariadb
   - Database: panel
   - User: pterodactyl
   - Port: 3306
{{- end }}

{{- if .Values.redis.enabled }}
3. Redis is deployed as a separate pod in the same namespace.
   - Service: {{ include "pterodactyl.fullname" . }}-redis
   - Port: 6379
{{- end }}

{{- if or .Values.mariadb.enabled .Values.redis.enabled }}
4. To configure passwords:
   - For MariaDB user password: set pterodactyl.database.password or use a secretKeyRef
   - For MariaDB root password: set mariadb.rootPassword or use a secretKeyRef
{{- if .Values.redis.enabled }}
   - For Redis password: set pterodactyl.redis.password or use a secretKeyRef
{{- end }}
   Example using a single secret:
   kubectl create secret generic pterodactyl-secrets \
     --from-literal=mariadb-root-password=<root_password> \
     --from-literal=mariadb-password=<user_password>
{{- if .Values.redis.enabled }}
     --from-literal=redis-password=<redis_password>
{{- end }}
   Then configure in values.yaml:
   pterodactyl:
     database:
       password:
         secretKeyRef:
           name: pterodactyl-secrets
           key: mariadb-password
   mariadb:
     rootPassword:
       secretKeyRef:
         name: pterodactyl-secrets
         key: mariadb-root-password
{{- if .Values.redis.enabled }}
   pterodactyl:
     redis:
       password:
         secretKeyRef:
           name: pterodactyl-secrets
           key: redis-password
{{- end }}
{{- end }}
